    1. Minkowski Distance
    def Minkowski(i,j,r):
    if i != j:
        hasil = 0
        for t in df:
            rumus = (df[t][i]-df[t][j])**r
            hasil = hasil + rumus
        h = 1
        if hasil>=0:
            h=1
        else:
            h=-1
        hasil = math.pow(hasil*h, 1/r)
        if h == 1:
            h = -1
        else:
            h = 1
        return hasil*h
    else:
        return 0

for xx in range(len(df)-1):

    1. Euclidean distance, m=2

    def Euclidean(i,j):
    if i != j:
        hasil = 0
        for fitur in df:
            rumus = (df[fitur][i]-df[fitur][j])**2
            hasil += rumus
        hasil = math.sqrt(hasil)
        return hasil
    else:
        return 0
